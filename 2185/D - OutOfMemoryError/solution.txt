import sys
from collections import defaultdict, Counter, deque
from bisect import bisect_left, bisect_right
input = sys.stdin.readline
 
def a_n():
    return int(input())
 
def a_s():
    return input().strip()
 
def a_map():
    return map(int, input().split())
 
def a_nums():
    return list(map(int, input().split()))
 
for _ in range(int(input())):
    n, m, h = a_map()
    nums = a_nums()
    original = nums[:]
    ops = []
    for _ in range(m):
        ops.append(a_nums())
    
    modified = []
    for b, c in ops:
        idx = b-1
        nums[idx] = nums[idx] + c
        modified.append(idx)
        
        if nums[idx] > h:
            for idx in modified:
                nums[idx] = original[idx]
            modified = []
    print(*nums)
